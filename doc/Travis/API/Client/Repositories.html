def build!(build_id)
  @options[:build_id] = build_id.to_s
  builds_cache().delete_if {|b| b.id == @options[:build_id]}
  build = Entity::Build.new(results_for(REPOSITORY_BUILD_PATH), @options[:owner], @options[:name])
  builds_cache() << build
  build
end
